Создать базу данных db_books.
CREATE DATABASE db_books;

Выбрать ее в качестве базы данных по умолчанию
USE db_books

Создать таблицу tbl_topics содержащую следующий набор столбцов
CREATE TABLE 
    `tbl_topics` (
        `topic_id` INT(127) NOT NULL UNIQUE,
        `topic_name` VARCHAR(100) UNIQUE);


Структура БД 
describe tbl_topics;

"Создать таблицу tbl_books для хранения данных о книгах, содержащую следующий набор столбцов:"
CREATE TABLE 
    `tbl_books` (
        `b_id` CHAR(17) NOT NULL PRIMARY KEY,
        `b_name` VARCHAR(255) NOT NULL,
        `b_author` VARCHAR(255),
        `b_topic` INT NOT NULL,
        `b_price` DECIMAL(5, 2) UNSIGNED);

Увеличить стоимость всех книг на 10%
UPDATE tbl_books SET b_price= b_price+(b_price*10)/100 WHERE b_price > 0;

Удалить все книги, стоимость которых неопределенна.
DELETE from tbl_books where b_price < 1;

Объединить таблицы
SELECT tbl_books.b_name,
       tbl_books.b_author, 
       tbl_books.b_price, 
       tbl_topics.topic_name  
       FROM tbl_books 
       LEFT JOIN tbl_topics on tbl_books.b_topic = tbl_topics.topic_id;
